<?php
function sumEvenNumbers($numbers) {
    $sum = 0;
    foreach ($numbers as $num) {
        if ($num % 2 == 0) {  // check if the number is even
            $sum += $num;     // add it to the sum
        }
    }
    return $sum;
}

// Example usage
$array = [1, 2, 3, 4, 5, 6];
echo "Sum of even numbers: " . sumEvenNumbers($array);  // Output: 12
?>
